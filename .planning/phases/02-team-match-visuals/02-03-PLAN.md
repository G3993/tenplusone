    **generative.ts**: Pure function mapping match data to visual parameters.

    ```typescript
    export interface MatchStats {
      homeScore: number;
      awayScore: number;
      possession: [number, number];     // [home%, away%]
      shots: [number, number];
      shotsOnTarget: [number, number];
      corners: [number, number];
      fouls: [number, number];
      yellowCards: [number, number];
      redCards: [number, number];
    }

    export interface GenerativeConfig {
      gridScale: number;                // 0.3-1.5 range
      gridRotation: number;             // 0-360 degrees
      shapeMode: 'cube' | 'sphere';
      darkMode: boolean;
      fluidIntensity: number;           // 0-1, from total goals
      fluidColor: [number, number, number]; // winning team's primary color
      splatCount: number;               // total shots
      logoBlend: number;                // 0-1, home vs away possession
      transitionStyle: 'wave' | 'spiral' | 'random' | 'rows' | 'columns' | 'diagonal';
      matchTitle: string;               // "Germany vs France"
      score: string;                    // "2 - 1"
      date: string;                     // ISO date string
    }

    export function matchToGenerativeConfig(
      stats: MatchStats,
      homeTeam: TeamData,
      awayTeam: TeamData,
      matchDate?: string
    ): GenerativeConfig
    ```

    Mapping rules (from research):
    - `gridScale`: 0.8 + (totalGoals * 0.05), clamped to [0.3, 1.5]
    - `gridRotation`: (possession[0] / 100) * 360
    - `shapeMode`: totalGoals > 4 ? 'sphere' : 'cube'
    - `darkMode`: homeScore > awayScore (home team dominant = dark)
    - `fluidIntensity`: Math.min(totalGoals * 0.15, 1.0)
    - `fluidColor`: winner's primaryColor (home if homeScore >= awayScore)
    - `splatCount`: shots[0] + shots[1]
    - `logoBlend`: possession[0] / 100
    - `transitionStyle`: goalDiff === 0 ? 'wave' : goalDiff === 1 ? 'spiral' : goalDiff >= 3 ? 'diagonal' : 'random'
    - `matchTitle`: `${homeTeam.name} vs ${awayTeam.name}`
    - `score`: `${homeScore} - ${awayScore}`
    - `date`: matchDate || new Date().toISOString().split('T')[0]
